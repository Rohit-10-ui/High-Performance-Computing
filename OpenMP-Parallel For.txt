#include<stdio.h>
#include<stdlib.h>
#include<omp.h>
#include<time.h>

int main(){
   int N;
   printf("**** Sum of Two Array using Parallel for ****\n");
   printf("Enter number of elements in Array: ");
   scanf("%d",&N);
   int a[N],b[N],c[N];
   srand(time(NULL));
   for(int i=0;i<N;i++){
      a[i]=rand()%50;
      b[i]=rand()%50;
   }
   printf("A: ");
   for(int i=0;i<N;i++)
      printf("%d ",a[i]);
   printf("\n");
   printf("B: ");
   for(int i=0;i<N;i++)
      printf("%d ",b[i]);
   printf("\n");
   #pragma omp parallel for
   for(int i=0;i<N;i++)
      c[i]=a[i]+b[i];
   printf("Result: ");
   for(int i=0;i<N;i++)
      printf("%d ",c[i]);
   printf("\n");
}

#include <stdio.h>
#include<stdlib.h>
#include <omp.h>
#include<time.h>

int main() {
    int N;
    srand(time(NULL));
    printf("******** Mirror Matrix Using Parallel for ********");
    printf("Enter size of matrix(N): ");
    scanf("%d",&N);
    int A[N][N],mirror[N][N];
    for(int i=0;i<N;i++){
       for(int j=0;j<N;j++){
 A[i][j]=rand()%50;
       }
    }
    printf("Matrix:\n");
    for(int i=0;i<N;i++){
       for(int j=0;j<N;j++){
 printf("%d ",A[i][j]);
       }
       printf("\n");
    }
    #pragma omp parallel for collapse(2)
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            mirror[i][j] = A[i][N - j - 1];
        }
    }

    printf("Original Matrix:\n");
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++)
            printf("%d ", A[i][j]);
        printf("\n");
    }

    printf("\nMirror Matrix:\n");
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++)
            printf("%d ", mirror[i][j]);
        printf("\n");
    }

    return 0;
}

#include <stdio.h>
#include <omp.h>
#include<time.h>
#include<stdlib.h>

int  main() {
    int N;
    printf("**** Running Minimum using Parallel For ****\n");
    printf("Enter number of Elements in Array: ");
    scanf("%d",&N);
    srand(time(NULL));
    int arr[N];
    for(int i=0;i<N;i++)
       arr[i]=rand()%50;
    int running_min[N];
    int global_min = arr[0];
    #pragma omp parallel for
    for (int i = 0; i < N; i++) {
        int local_min = arr[i];
        for (int j = 0; j <= i; j++) {
            if (arr[j] < local_min) {
                local_min = arr[j];
            }
        }
        running_min[i] = local_min;
    }
    printf("Original Array:\n");
    for (int i = 0; i < N; i++)
        printf("%d ", arr[i]);
    printf("\nRunning Minimum:\n");
    for (int i = 0; i < N; i++)
        printf("%d ", running_min[i]);
    printf("\n");
    return 0;
}
