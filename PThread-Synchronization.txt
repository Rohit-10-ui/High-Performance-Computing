#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <pthread.h>

#define BUFFER_SIZE 10
#define WORD_LEN 50
#define THREADS 3

char *dictionary[] = {"arch","bios","orange","linux","windows","compiler","interpreter","nginx","severless"};
int dict_size = 9;

char buffer[BUFFER_SIZE][WORD_LEN];
int count = 0;
int done = 0;

pthread_mutex_t lock;
pthread_cond_t not_empty;
pthread_cond_t not_full;

int spell_check(char *word)
{
    for(int i = 0; i < dict_size; i++)
    {
        if(strcmp(dictionary[i], word) == 0)
            return 1;
    }
    return 0;
}

void* spell_checker(void *arg)
{
    char word[WORD_LEN];
    while(1)
    {
        pthread_mutex_lock(&lock);
        while(count == 0 && !done)
            pthread_cond_wait(&not_empty, &lock);

        if(count == 0 && done) {
            pthread_mutex_unlock(&lock);
            break;
        }

        strcpy(word, buffer[count - 1]);
        count--;
        pthread_cond_signal(&not_full);
        pthread_mutex_unlock(&lock);

        int correct = spell_check(word);

        pthread_mutex_lock(&lock);
        if(correct)
            printf("%s -> Correct\n", word);
        else
            printf("%s -> Incorrect\n", word);
        pthread_mutex_unlock(&lock);
    }
    return NULL;
}

int main()
{
    pthread_t workers[THREADS];
    pthread_mutex_init(&lock, NULL);
    pthread_cond_init(&not_empty, NULL);
    pthread_cond_init(&not_full, NULL);
    printf("*** Spell Checker Using Pthread Synchronization ***\n");
    /*printf("Dictionary available:\n");
    for(int i = 0; i < dict_size; i++)
        printf(" - %s\n", dictionary[i]);
    printf("\n");*/

    int n;
    printf("Enter number of words to check: ");
    scanf("%d", &n);

    for(int i = 0; i < THREADS; i++)
        pthread_create(&workers[i], NULL, spell_checker, NULL);

    printf("Enter words one by one:\n");
    for(int i = 0; i < n; i++)
    {
        char word[WORD_LEN];
        scanf("%s", word);

        pthread_mutex_lock(&lock);
        while(count == BUFFER_SIZE)
            pthread_cond_wait(&not_full, &lock);
        strcpy(buffer[count++], word);
        pthread_cond_signal(&not_empty);
        pthread_mutex_unlock(&lock);
    }
    pthread_mutex_lock(&lock);
    done = 1;
    pthread_cond_broadcast(&not_empty);
    pthread_mutex_unlock(&lock);

    for(int i = 0; i < THREADS; i++)
        pthread_join(workers[i], NULL);

    pthread_mutex_destroy(&lock);
    pthread_cond_destroy(&not_empty);
    pthread_cond_destroy(&not_full);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>
int pending_submissions = 0;
int max_limit = 0;
int evaluation_done = 0;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t cond_evaluator = PTHREAD_COND_INITIALIZER;
pthread_cond_t cond_submitter = PTHREAD_COND_INITIALIZER;
void* student_submitter(void* arg) {
    char input;
    printf("\n--- COMMANDS ---\nType 's' to SUBMIT a script or 'q' to End.\n ----------------\n");
    while (1) {
        scanf(" %c", &input);
        if (input == 's' || input == 'S') {
            pthread_mutex_lock(&mutex);        
            while (pending_submissions >= max_limit) {
                printf("[System] LIMIT REACHED (%d). Waiting for evaluator...\n", max_limit);
                pthread_cond_wait(&cond_submitter, &mutex);
            }          
            pending_submissions++;
            printf("[Student] Submission successful! Pending: %d\n", pending_submissions);
            pthread_cond_signal(&cond_evaluator);
            pthread_mutex_unlock(&mutex);
        }
else if (input == 'q' || input =='Q'){
  printf("Done..\n");
  exit(1);
}
else {
            printf("Invalid command. Use 's' to submit.\n");
        }
    }
    return NULL;
}

void* evaluator(void* arg) {
    while (1) {
        pthread_mutex_lock(&mutex);
        while (pending_submissions == 0) {
            pthread_cond_wait(&cond_evaluator, &mutex);
        }    
        printf("[Evaluator] Starting evaluation...\n");
        pthread_mutex_unlock(&mutex);
        sleep(3);        
        pthread_mutex_lock(&mutex);
        pending_submissions--;
        evaluation_done = 1;
        printf("[Evaluator] Finished. Remaining in queue: %d\n", pending_submissions);      
        pthread_cond_signal(&cond_submitter);
        pthread_mutex_unlock(&mutex);
    }
    return NULL;
}

void* monitor(void* arg) {
    int last_seen = -1;
    while (1) {
        pthread_mutex_lock(&mutex);
        if (pending_submissions != last_seen) {
            printf("\n>>> [Monitor] Queue Status: %d scripts waiting.\n", pending_submissions);
            last_seen = pending_submissions;
        }
        pthread_mutex_unlock(&mutex);
        usleep(500000);
    }
    return NULL;
}

void* result_publisher(void* arg) {
    while (1) {
        pthread_mutex_lock(&mutex);
        if (evaluation_done) {
            printf("[Publisher] Result published for an evaluated script.\n");
            evaluation_done = 0;
        }
        pthread_mutex_unlock(&mutex);
        usleep(100000);
    }
    return NULL;
}

int main() {
    pthread_t t1, t2, t3, t4;
    printf("**** Printer Submit Queue using Pthread Semaphore ****\n");
    printf("Enter the MAX capacity of the submission queue: ");
    scanf("%d", &max_limit);

    pthread_create(&t1, NULL, student_submitter, NULL);
    pthread_create(&t2, NULL, evaluator, NULL);
    pthread_create(&t3, NULL, monitor, NULL);
    pthread_create(&t4, NULL, result_publisher, NULL);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    pthread_join(t3, NULL);
    pthread_join(t4, NULL);

    return 0;
}
